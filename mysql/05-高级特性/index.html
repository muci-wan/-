    05 — 高级特性¶

    5.1 存储引擎¶

    InnoDB vs MyISAM¶

    特性 InnoDB MyISAM
    事务支持 支持 不支持
    行锁 支持 表锁
    外键 支持 不支持
    崩溃恢复 支持（redo log） 不支持
    MVCC 支持 不支持
    全文索引 5.6+ 支持 支持
    空间索引 5.7+ 支持 支持
    压缩 5.7+ 支持 支持
    适用场景 高并发 OLTP 只读查询 / 日志

    MySQL 8.0 已废弃 MyISAM，新项目一律用 InnoDB。

    -- 查看支持的引擎
SHOW ENGINES;

-- 指定引擎
CREATE TABLE t1 (id INT) ENGINE=InnoDB;

-- 修改引擎
ALTER TABLE t1 ENGINE=InnoDB;

-- 查看表使用的引擎
SHOW TABLE STATUS WHERE Name = 'student';

    InnoDB 核心特性¶

    • Buffer Pool：缓存数据和索引，提高读写性能
    • Change Buffer：缓存二级索引的变更，延迟写入
    • Adaptive Hash Index：自动对热点页创建 Hash 索引
    • Double Write Buffer：防止部分页写入崩溃
    • Redo Log：保证持久性
    • Undo Log：支持事务回滚和 MVCC
    -- Buffer Pool 大小（一般为物理内存的 50%-70%）
SET GLOBAL innodb_buffer_pool_size = 2147483648;  -- 2GB

-- 查看状态
SHOW STATUS LIKE 'Innodb_buffer_pool%';

    5.2 视图¶

    创建和使用¶

    -- 创建视图
CREATE VIEW v_student_score AS
SELECT s.id, s.name, sc.subject, sc.score
FROM student s
JOIN score sc ON s.id = sc.student_id;

-- 查询视图（和查表一样）
SELECT * FROM v_student_score WHERE score > 90;

-- 可更新视图（单表、无聚合）
CREATE VIEW v_active_student AS
SELECT * FROM student WHERE status = 1;

-- WITH CHECK OPTION：防止通过视图插入不符合条件的数据
CREATE VIEW v_male_student AS
SELECT * FROM student WHERE gender = '男'
WITH CHECK OPTION;

-- 查看视图定义
SHOW CREATE VIEW v_student_score;

-- 删除视图
DROP VIEW v_student_score;

    视图的优缺点¶

    • 优点：简化复杂查询、隐藏表结构、提供安全层
    • 缺点：不能有索引（MySQL）、性能可能不如实际表、依赖原表结构

    5.3 存储过程¶

    -- 修改分隔符（解决 ; 冲突）
DELIMITER //

-- 创建存储过程
CREATE PROCEDURE get_student_scores(IN sid INT)
BEGIN
    SELECT s.name, sc.subject, sc.score
    FROM student s
    JOIN score sc ON s.id = sc.student_id
    WHERE s.id = sid;
END //

DELIMITER ;

-- 调用存储过程
CALL get_student_scores(1);

-- 带输出参数的存储过程
DELIMITER //
CREATE PROCEDURE get_avg_score(IN sid INT, OUT avg_sc DECIMAL(5,2))
BEGIN
    SELECT AVG(score) INTO avg_sc
    FROM score WHERE student_id = sid;
END //
DELIMITER ;

-- 调用并获取输出
CALL get_avg_score(1, @result);
SELECT @result;

-- 查看存储过程
SHOW PROCEDURE STATUS WHERE Db = 'mydb';
SHOW CREATE PROCEDURE get_student_scores;

-- 删除
DROP PROCEDURE IF EXISTS get_student_scores;

    5.4 函数¶

    DELIMITER //

CREATE FUNCTION calc_level(score DECIMAL(5,2))
RETURNS VARCHAR(10)
DETERMINISTIC
BEGIN
    DECLARE level VARCHAR(10);
    IF score >= 90 THEN SET level = 'A';
    ELSEIF score >= 80 THEN SET level = 'B';
    ELSEIF score >= 70 THEN SET level = 'C';
    ELSEIF score >= 60 THEN SET level = 'D';
    ELSE SET level = 'F';
    END IF;
    RETURN level;
END //

DELIMITER ;

-- 使用函数
SELECT name, score, calc_level(score) AS level FROM score;

    5.5 触发器¶

    -- 创建触发器（在插入 score 后更新 student 的更新时间）
DELIMITER //
CREATE TRIGGER trg_score_insert
AFTER INSERT ON score
FOR EACH ROW
BEGIN
    UPDATE student SET updated_at = NOW() WHERE id = NEW.student_id;
END //
DELIMITER ;

-- 查看触发器
SHOW TRIGGERS;

-- 删除触发器
DROP TRIGGER IF EXISTS trg_score_insert;

-- 触发时机
-- BEFORE INSERT / AFTER INSERT
-- BEFORE UPDATE / AFTER UPDATE
-- BEFORE DELETE / AFTER DELETE
-- NEW 和 OLD 的可用性：
-- INSERT → NEW 可用，OLD 为 NULL
-- UPDATE → NEW 和 OLD 都可用
-- DELETE → NEW 为 NULL，OLD 可用

    5.6 事件（定时任务）¶

    -- 开启事件调度器
SET GLOBAL event_scheduler = ON;

-- 创建定时事件（每天清理 30 天前的日志）
CREATE EVENT clean_old_logs
ON SCHEDULE EVERY 1 DAY
STARTS '2026-06-21 02:00:00'
DO
    DELETE FROM operation_log
    WHERE created_at < DATE_SUB(NOW(), INTERVAL 30 DAY);

-- 一次性事件
CREATE EVENT one_time_job
ON SCHEDULE AT '2026-06-21 10:00:00'
DO
    INSERT INTO log (msg) VALUES ('一次性任务执行');

-- 查看事件
SHOW EVENTS;

-- 删除
DROP EVENT IF EXISTS clean_old_logs;
